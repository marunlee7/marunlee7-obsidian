## 📕 작업노트
```dataview
table without id
link(file.path, regexreplace(file.name, "^(.{4}).*", "$1")) AS "제목", 
작업노트
FROM ""
WHERE 작업노트 != null
WHERE contains(file.tags, "정민/성대중처세어록")
sort file.mtime desc
```

## 📕 MOC
```dataview
TABLE WITHOUT ID
link(file.path, substring(regexreplace(file.name, "^(.*?) -.*", "$1"), 0, 25)) AS "제목",
choice(checkbox, "❗", "") AS "❗",
dateformat(created, "yyyy-MM-dd") as "시작일",
dateformat(modified, "yyyy-MM-dd") as "종료일"
FROM "" 
WHERE contains(file.tags, "정민/성대중처세어록")
and !contains(file.folder, "90-Settings")
and !contains(file.folder, "10-프로젝트")
and !contains(file.folder, "20-에어리어")
and !contains(file.folder, "40-아카이브")
AND !contains(file.tags, "moc") 
AND !contains(file.tags, "toc")
sort file.name desc
//sort file.mtime desc
//sort file.name asc
sort file.name desc
```
